43 
44  ! Parameters

46 
47  ! Local Variables
48  real(real64) :: eps, epsmch, h, temp, f1, f0
49 
50  ! Initialization
51  epsmch = epsilon(epsmch)
52  eps = sqrt(epsmch)
53 
54  ! Process
55  if (this%is_derivative_defined()) then
56  ! Use the user-defined routine to compute the derivative
57  df = this%m_diff(x)
58  else
59  ! Compute the derivative via a forward difference
60  h = eps * abs(x)
61  if (h < epsmch) h = eps
62  temp = x + h
63  f1 = this%fcn(temp)
64  if (present(f)) then
65  f0 = f
66  else
67  f0 = this%fcn(x)
68  end if
69  df = (f1 - f0) / h
70  end if
71  end function
