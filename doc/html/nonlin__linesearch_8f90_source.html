592  pure function min_backtrack_search(mode, f0, f, f1, alam, alam1, &
593  slope) result(lam)
594  ! Arguments
595  integer(int32), intent(in) :: mode
596  real(real64), intent(in) :: f0, f, f1, alam, alam1, slope
597  real(real64) :: lam
598 
599  ! Parameters
600  real(real64), parameter :: zero = 0.0d0
601  real(real64), parameter :: p5 = 0.5d0
602  real(real64), parameter :: two = 2.0d0
603  real(real64), parameter :: three = 3.0d0
604 
605  ! Local Variables
606  real(real64) :: rhs1, rhs2, a, b, disc
607 
608  ! Process
609  if (mode == 1) then
610  ! Use a quadratic function approximation
611  lam = -slope / (two * (f - f0 - slope))
612  else
613  ! Use a cubic function
614  rhs1 = f - f0 - alam * slope
615  rhs2 = f1 - f0 - alam1 * slope
616  a = (rhs1 / alam**2 - rhs2 / alam1**2) / (alam - alam1)
617  b = (-alam1 * rhs1 / alam**2 + alam * rhs2 / alam1**2) / &
618  (alam - alam1)
619  if (a == zero) then
620  lam = -slope / (two * b)
621  else
622  disc = b**2 - three * a * slope
623  if (disc < zero) then
624  lam = p5 * alam
625  else if (b <= zero) then
626  lam = (-b + sqrt(disc)) / (three * a)
627  else
628  lam = -slope / (b + sqrt(disc))
629  end if
630  end if
631  if (lam > p5 * alam) lam = p5 * alam
632  end if
633  end function
