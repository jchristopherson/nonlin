46 ! ------------------------------------------------------------------------------
47  module subroutine vfh_jac_fcn(this, x, jac, fv, work, olwork, err)
48  ! Arguments
49  class(vecfcn_helper), intent(in) :: this
50  real(real64), intent(inout), dimension(:) :: x
51  real(real64), intent(out), dimension(:,:) :: jac
52  real(real64), intent(in), dimension(:), optional, target :: fv
53  real(real64), intent(out), dimension(:), optional, target :: work
54  integer(int32), intent(out), optional :: olwork, err
55 
56  ! Parameters
57  real(real64), parameter :: zero = 0.0d0
58 
59  ! Local Variables
60  integer(int32) :: j, m, n, lwork, flag
61  real(real64) :: eps, epsmch, h, temp
62  real(real64), pointer, dimension(:) :: fptr, f1ptr
63  real(real64), allocatable, target, dimension(:) :: wrk
64 
65  ! Initialization
66  if (present(err)) err = 0
67  ! m = this%m_nfcn
68  ! n = this%m_nvar
69  m = this%get_equation_count()
70  n = this%get_variable_count()
71 
72  ! Input Checking
73  flag = 0
74  if (size(x) /= n) then
75  flag = 2
76  else if (size(jac, 1) /= m .or. size(jac, 2) /= n) then
77  flag = 3
78  end if
79  if (flag /= 0) then
80  ! ERROR: Incorrectly sized input arrays
81  if (present(err)) err = flag
82  return
83  end if
84 
85  ! Process
86  if (.not.this%is_fcn_defined()) return
87  if (associated(this%m_jac)) then
88  ! Workspace Query
89  if (present(olwork)) then
90  olwork = 0
91  return
92  end if
93 
94  ! Call the user-defined Jacobian routine
95  call this%m_jac(x, jac)
96  else
97  ! Compute the Jacobian via finite differences
98  if (present(fv)) then
99  lwork = m
100  else
101  lwork = 2 * m
102  end if
103 
104  if (present(olwork)) then
105  ! The user is just making a workspace query. Simply return the
106  ! workspace length, and exit the routine.
107  olwork = lwork
108  return
109  end if
110 
111  ! Local Memory Allocation
112  if (present(work)) then
113  if (size(work) < lwork) then
114  ! ERROR: Workspace is too small
115  if (present(err)) err = 5
116  return
117  end if
118  f1ptr => work(1:m)
119  if (present(fv)) then
120  if (size(fv) < m) then
121  ! ERROR: Function vector too small
122  if (present(err)) err = 4
123  return
124  end if
125  fptr => fv(1:m)
126  else
127  fptr => work(m+1:2*m)
128  call this%fcn(x, fptr)
129  end if
130  else
131  allocate(wrk(lwork), stat = flag)
132  if (flag /= 0) then
133  ! ERROR: Memory issues
134  if (present(err)) err = -1
135  return
136  end if
137  f1ptr => wrk(1:m)
138  if (present(fv)) then
139  fptr => fv(1:m)
140  else
141  fptr => wrk(m+1:2*m)
142  call this%fcn(x, fptr)
143  end if
144  end if
145 
146  ! Establish step size factors
147  epsmch = epsilon(epsmch)
148  eps = sqrt(epsmch)
149 
150  ! Compute the derivatives via finite differences
151  do j = 1, n
152  temp = x(j)
153  h = eps * abs(temp)
154  if (h == zero) h = eps
155  x(j) = temp + h
156  call this%fcn(x, f1ptr)
157  x(j) = temp
158  jac(:,j) = (f1ptr - fptr) / h
159  end do
160  end if
161  end subroutine
