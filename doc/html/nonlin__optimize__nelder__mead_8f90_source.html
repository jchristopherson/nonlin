6  module subroutine nm_solve(this, fcn, x, fout, ib, err)
7  ! Arguments
8  class(nelder_mead), intent(inout) :: this
9  class(fcnnvar_helper), intent(in) :: fcn
10  real(real64), intent(inout), dimension(:) :: x
11  real(real64), intent(out), optional :: fout
12  type(iteration_behavior), optional :: ib
13  class(errors), intent(inout), optional, target :: err
14 
15  ! Parameters
16  real(real64), parameter :: zero = 0.0d0
17  real(real64), parameter :: negone = -1.0d0
18  real(real64), parameter :: half = 0.5d0
19  real(real64), parameter :: two = 2.0d0
20 
21  ! Local Variables
22  logical :: buildSimplex, fcnvrg
23  integer(int32) :: i, ihi, ilo, ihi2, ndim, npts, flag, neval, iter, &
24  maxeval
25  real(real64) :: ftol, rtol, ftry, fsave, fval, swp
26  real(real64), allocatable, dimension(:) :: f, pcent, pmin, work
27  class(errors), pointer :: errmgr
28  type(errors), target :: deferr
29  character(len = 256) :: errmsg
30 
31  ! Initialization
32  ndim = fcn%get_variable_count()
33  npts = ndim + 1
34  buildsimplex = .true.
35  maxeval = this%get_max_fcn_evals()
36  ftol = this%get_tolerance()
37  iter = 0
38  neval = 0
39  fcnvrg = .false.
40  if (present(ib)) then
41  ib%iter_count = iter
42  ib%fcn_count = neval
43  ib%jacobian_count = 0
44  ib%gradient_count = 0
45  ib%converge_on_fcn = fcnvrg
46  ib%converge_on_chng = .false.
47  ib%converge_on_zero_diff = .false.
48  end if
49  if (present(err)) then
50  errmgr => err
51  else
52  errmgr => deferr
53  end if
54 
55  ! Input Check
56  if (.not.fcn%is_fcn_defined()) then
57  ! ERROR: No function is defined
58  call errmgr%report_error("nm_solve", &
59  "No function has been defined.", &
60  nl_invalid_operation_error)
61  return
62  end if
63  if (size(x) /= ndim) then
64  write(errmsg, '(AI0AI0A)') &
65  "It was expected to receive a coordinate vector of length ", &
66  ndim, " , but a vector of length ", size(x), " was received."
67  call errmgr%report_error("nm_solve", trim(errmsg), &
68  nl_invalid_input_error)
69  return
70  end if
71 
72  ! Ensure that if an initial simplex was defined, that it is
73  ! appropriately sized. If not, simply create a new simplex of the
74  ! appropriate size.
75  if (allocated(this%m_simplex)) then
76  ! This matrix must be NDIM-by-NPTS
77  if (size(this%m_simplex, 1) /= ndim .or. &
78  size(this%m_simplex, 2) /= npts) then
79  deallocate(this%m_simplex)
80  buildsimplex = .true.
81  else
82  ! The simplex is appropriately sized
83  buildsimplex = .false.
84  end if
85  end if
86 
87  ! Local Memory Allocation
88  allocate(f(npts), stat = flag)
89  if (flag == 0) allocate(pcent(ndim), stat = flag)
90  if (flag == 0) allocate(pmin(ndim), stat = flag)
91  if (flag == 0) allocate(work(ndim), stat = flag)
92  if (buildsimplex .and. flag == 0) allocate(this%m_simplex(ndim, npts))
93  if (flag /= 0) then
94  ! ERROR: Out of memory
95  call errmgr%report_error("nm_solve", &
96  "Insufficient memory available.", nl_out_of_memory_error)
97  return
98  end if
99 
100  ! Define the initial simplex, if needed
101  if (buildsimplex) then
102  this%m_simplex(:,1) = x
103  do i = 2, npts
104  this%m_simplex(:,i) = x
105  end do
106  do i = 1, ndim
107  this%m_simplex(i,i+1) = this%m_simplex(i,i+1) + this%m_initSize
108  end do
109  end if
110 
111  ! Evaluate the function at each vertex of the simplex
112  do i = 1, npts
113  f(i) = fcn%fcn(this%m_simplex(:,i))
114  end do
115  neval = npts
116  fval = f(1)
117 
118  do i = 1, ndim
119  pcent(i) = sum(this%m_simplex(i,:))
120  end do
121 
122  ! Main Loop
123  flag = 0 ! Used to check for convergence errors
124  do
125  ! Update the iteration counter
126  iter = iter + 1
127 
128  ! Determine the characteristics of each vertex
129  ilo = 1
130  if (f(1) > f(2)) then
131  ihi = 1
132  ihi2 = 2
133  else
134  ihi = 2
135  ihi2 = 1
136  end if
137  do i = 1, npts
138  if (f(i) <= f(ilo)) ilo = i
139  if (f(i) > f(ihi)) then
140  ihi2 = ihi
141  ihi = i
142  else if (f(i) > f(ihi2)) then
143  if (i /= ihi) ihi2 = i
144  end if
145  end do
146 
147  ! Check for convergence. Nelder and Mead recommend using the
148  ! following convergence test: sqrt(sum(f - favg)**2 / n); however,
149  ! it seems that a sufficient check may be made using only the
150  ! extreme function values of the simplex (highest and lowest valued
151  ! points).
152  rtol = abs(f(ihi) - f(ilo))
153  if (rtol < ftol) then
154  swp = f(1)
155  f(1) = f(ilo)
156  f(ilo) = swp
157  do i = 1, ndim
158  swp = this%m_simplex(i,1)
159  this%m_simplex(i,1) = this%m_simplex(i,ilo)
160  this%m_simplex(i,ilo) = swp
161  x(i) = this%m_simplex(i,1)
162  end do
163  fval = f(1)
164  fcnvrg = .true.
165  exit
166  end if
167 
168  ! Start of a new iteration by reflecting the simplex at its largest
169  ! point.
170  ftry = this%extrapolate(fcn, f, pcent, ihi, negone, neval, work)
171  if (ftry <= f(ilo)) then
172  ! The result of the reflection is better than the current
173  ! best point. As a result, try a factor of 2 in the reflected
174  ! direction. Again, the highest point is of interest.
175  ftry = this%extrapolate(fcn, f, pcent, ihi, two, neval, work)
176  else if (ftry >= f(ihi2)) then
177  ! The reflected point is worse than the second highest, so look
178  ! for an intermediate lower point (contract the simplex)
179  fsave = f(ihi)
180  ftry = this%extrapolate(fcn, f, pcent, ihi, half, neval, work)
181  if (ftry >= fsave) then
182  ! Cannot improve on the high point. Try to contract around
183  ! the low point.
184  do i = 1, npts
185  if (i /= ilo) then
186  pcent = half * (this%m_simplex(:,i) + &
187  this%m_simplex(:,ilo))
188  this%m_simplex(:,i) = pcent
189  f(i) = fcn%fcn(pcent)
190  end if
191  end do
192  neval = neval + npts
193  do i = 1, ndim
194  pcent(i) = sum(this%m_simplex(i,:))
195  end do
196  end if
197  end if
198 
199  ! Print iteration status
200  if (this%get_print_status()) then
201  print *, ""
202  print '(AI0)', "Iteration: ", iter
203  print '(AI0)', "Function Evaluations: ", neval
204  print '(AE8.3)', "Function Value: ", fval
205  print '(AE8.3)', "Convergence Parameter: ", rtol
206  end if
207 
208  ! Ensure we haven't made too many function evaluations
209  if (neval >= maxeval) then
210  flag = 1
211  exit
212  end if
213  end do
214 
215  ! Report out iteration statistics
216  if (present(ib)) then
217  ib%iter_count = iter
218  ib%fcn_count = neval
219  ib%jacobian_count = 0
220  ib%gradient_count = 0
221  ib%converge_on_fcn = fcnvrg
222  ib%converge_on_chng = .false.
223  ib%converge_on_zero_diff = .false.
224  end if
225 
226  ! Get the function value at the computed minimum
227  if (present(fout)) fout = fval
228 
229  ! Check for convergence issues
230  if (flag /= 0) then
231  write(errmsg, '(AI0AE8.3AE8.3)') &
232  "The algorithm failed to converge." // new_line('c') // &
233  "Function evaluations performed: ", neval, new_line('c') // &
234  "Convergence Parameter: ", rtol, new_line('c') // &
235  "Convergence Criteria: ", ftol
236  call errmgr%report_error("nm_solve", trim(errmsg), &
237  nl_convergence_error)
238  end if
239  end subroutine

241 ! ------------------------------------------------------------------------------
242  module function nm_extrapolate(this, fcn, y, pcent, ihi, fac, neval, &
243  work) result(ytry)
244  ! Arguments
245  class(nelder_mead), intent(inout) :: this
246  class(fcnnvar_helper), intent(in) :: fcn
247  real(real64), intent(inout), dimension(:) :: y, pcent
248  integer(int32), intent(in) :: ihi
249  real(real64), intent(in) :: fac
250  integer(int32), intent(inout) :: neval
251  real(real64), intent(out), dimension(:) :: work
252  real(real64) :: ytry
253 
254  ! Parameters
255  real(real64), parameter :: one = 1.0d0
256 
257  ! Local Variables
258  integer(int32) :: i, ndim
259  real(real64) :: fac1, fac2
260 
261  ! Initialization
262  ndim = size(this%m_simplex, 1)
263 
264  ! Define a trial point
265  fac1 = (one - fac) / ndim
266  fac2 = fac1 - fac
267  do i = 1, ndim
268  work(i) = pcent(i) * fac1 - this%m_simplex(i,ihi) * fac2
269  end do
270 
271  ! Evaluate the function at the trial point, and then replace if the
272  ! trial provides an improvement
273  ytry = fcn%fcn(work)
274  neval = neval + 1
275  if (ytry < y(ihi)) then
276  y(ihi) = ytry
277  do i = 1, ndim
278  pcent(i) = pcent(i) + work(i) - this%m_simplex(i,ihi)
279  this%m_simplex(i,ihi) = work(i)
280  end do
281  end if
282  end function
