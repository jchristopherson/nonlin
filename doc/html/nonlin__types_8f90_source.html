574  subroutine vfh_jac_fcn(this, x, jac, fv, work, olwork, err)
575  ! Arguments
576  class(vecfcn_helper), intent(in) :: this
577  real(real64), intent(inout), dimension(:) :: x
578  real(real64), intent(out), dimension(:,:) :: jac
579  real(real64), intent(in), dimension(:), optional, target :: fv
580  real(real64), intent(out), dimension(:), optional, target :: work
581  integer(int32), intent(out), optional :: olwork, err
582 
583  ! Parameters
584  real(real64), parameter :: zero = 0.0d0
585 
586  ! Local Variables
587  integer(int32) :: j, m, n, lwork, flag
588  real(real64) :: eps, epsmch, h, temp
589  real(real64), pointer, dimension(:) :: fptr, f1ptr
590  real(real64), allocatable, target, dimension(:) :: wrk
591 
592  ! Initialization
593  if (present(err)) err = 0
594  ! m = this%m_nfcn
595  ! n = this%m_nvar
596  m = this%get_equation_count()
597  n = this%get_variable_count()
598 
599  ! Input Checking
600  flag = 0
601  if (size(x) /= n) then
602  flag = 2
603  else if (size(jac, 1) /= m .or. size(jac, 2) /= n) then
604  flag = 3
605  end if
606  if (flag /= 0) then
607  ! ERROR: Incorrectly sized input arrays
608  if (present(err)) err = flag
609  return
610  end if
611 
612  ! Process
613  if (.not.this%is_fcn_defined()) return
614  if (associated(this%m_jac)) then
615  ! Workspace Query
616  if (present(olwork)) then
617  olwork = 0
618  return
619  end if
620 
621  ! Call the user-defined Jacobian routine
622  call this%m_jac(x, jac)
623  else
624  ! Compute the Jacobian via finite differences
625  if (present(fv)) then
626  lwork = m
627  else
628  lwork = 2 * m
629  end if
630 
631  if (present(olwork)) then
632  ! The user is just making a workspace query. Simply return the
633  ! workspace length, and exit the routine.
634  olwork = lwork
635  return
636  end if
637 
638  ! Local Memory Allocation
639  if (present(work)) then
640  if (size(work) < lwork) then
641  ! ERROR: Workspace is too small
642  if (present(err)) err = 5
643  return
644  end if
645  f1ptr => work(1:m)
646  if (present(fv)) then
647  if (size(fv) < m) then
648  ! ERROR: Function vector too small
649  if (present(err)) err = 4
650  return
651  end if
652  fptr => fv(1:m)
653  else
654  fptr => work(m+1:2*m)
655  call this%fcn(x, fptr)
656  end if
657  else
658  allocate(wrk(lwork), stat = flag)
659  if (flag /= 0) then
660  ! ERROR: Memory issues
661  if (present(err)) err = -1
662  return
663  end if
664  f1ptr => wrk(1:m)
665  if (present(fv)) then
666  fptr => fv(1:m)
667  else
668  fptr => wrk(m+1:2*m)
669  call this%fcn(x, fptr)
670  end if
671  end if
672 
673  ! Establish step size factors
674  epsmch = epsilon(epsmch)
675  eps = sqrt(epsmch)
676 
677  ! Compute the derivatives via finite differences
678  do j = 1, n
679  temp = x(j)
680  h = eps * abs(temp)
681  if (h == zero) h = eps
682  x(j) = temp + h
683  call this%fcn(x, f1ptr)
684  x(j) = temp
685  jac(:,j) = (f1ptr - fptr) / h
686  end do
687  end if
688  end subroutine

834 ! ------------------------------------------------------------------------------
851  subroutine fnh_grad_fcn(this, x, g, fv, err)
852  ! Arguments
853  class(fcnnvar_helper), intent(in) :: this
854  real(real64), intent(inout), dimension(:) :: x
855  real(real64), intent(out), dimension(:) :: g
856  real(real64), intent(in), optional :: fv
857  integer(int32), intent(out), optional :: err
858 
859  ! Parameters
860  real(real64), parameter :: zero = 0.0d0
861 
862  ! Local Variables
863  integer(int32) :: j, n, flag
864  real(real64) :: eps, epsmch, h, temp, f, f1
865 
866  ! Initialization
867  if (present(err)) err = 0
868  ! n = this%m_nvar
869  n = this%get_variable_count()
870 
871  ! Input Checking
872  flag = 0
873  if (size(x) /= n) then
874  flag = 2
875  else if (size(g) /= n) then
876  flag = 3
877  end if
878  if (flag /= 0) then
879  ! ERROR: Incorrectly sized input arrays
880  if (present(err)) err = flag
881  return
882  end if
883 
884  ! Process
885  if (.not.this%is_fcn_defined()) return
886  if (this%is_gradient_defined()) then
887  ! Call the user-defined gradient routine
888  call this%m_grad(x, g)
889  else
890  ! Compute the gradient via finite differences
891  if (present(fv)) then
892  f = fv
893  else
894  f = this%fcn(x)
895  end if
896 
897  ! Establish step size factors
898  epsmch = epsilon(epsmch)
899  eps = sqrt(epsmch)
900 
901  ! Compute the derivatives
902  do j = 1, n
903  temp = x(j)
904  h = eps * abs(temp)
905  if (h == zero) h = eps
906  x(j) = temp + h
907  f1 = this%fcn(x)
908  x(j) = temp
909  g(j) = (f1 - f) / h
910  end do
911  end if
912  end subroutine
