250  subroutine qns_solve(this, fcn, x, fvec, ib, err)
251  ! Arguments
252  class(quasi_newton_solver), intent(inout) :: this
253  class(vecfcn_helper), intent(in) :: fcn
254  real(real64), intent(inout), dimension(:) :: x
255  real(real64), intent(out), dimension(:) :: fvec
256  type(iteration_behavior), optional :: ib
257  class(errors), intent(inout), optional, target :: err
258 
259  ! Parameters
260  real(real64), parameter :: zero = 0.0d0
261  real(real64), parameter :: half = 0.5d0
262  real(real64), parameter :: one = 1.0d0
263  real(real64), parameter :: factor = 1.0d2
264 
265  ! Local Variables
266  logical :: restart, xcnvrg, fcnvrg, gcnvrg, check
267  integer(int32) :: i, neqn, nvar, flag, lw1, lw2, lw3, neval, iter, &
268  maxeval, jcount, njac
269  real(real64), allocatable, dimension(:) :: work, tau, dx, df, fvold, &
270  xold, s
271  real(real64), allocatable, dimension(:,:) :: q, r, b
272  real(real64) :: test, f, fold, temp, ftol, xtol, gtol, &
273  stpmax, x2, xnorm, fnorm
274  type(iteration_behavior) :: lib
275  class(errors), pointer :: errmgr
276  type(errors), target :: deferr
277  character(len = 256) :: errmsg
278  class(line_search), allocatable :: ls
279 
280  ! Initialization
281  restart = .true.
282  xcnvrg = .false.
283  fcnvrg = .false.
284  gcnvrg = .false.
285  neqn = fcn%get_equation_count()
286  nvar = fcn%get_variable_count()
287  neval = 0
288  iter = 0
289  njac = 0
290  ftol = this%get_fcn_tolerance()
291  xtol = this%get_var_tolerance()
292  gtol = this%get_gradient_tolerance()
293  maxeval = this%get_max_fcn_evals()
294  if (present(ib)) then
295  ib%iter_count = iter
296  ib%fcn_count = neval
297  ib%jacobian_count = njac
298  ib%gradient_count = 0
299  ib%converge_on_fcn = fcnvrg
300  ib%converge_on_chng = xcnvrg
301  ib%converge_on_zero_diff = gcnvrg
302  end if
303  if (present(err)) then
304  errmgr => err
305  else
306  errmgr => deferr
307  end if
308  if (this%get_use_line_search()) then
309  if (.not.this%is_line_search_defined()) &
310  call this%set_default_line_search()
311  call this%get_line_search(ls)
312  end if
313 
314  ! Input Check
315  if (.not.fcn%is_fcn_defined()) then
316  ! ERROR: No function is defined
317  call errmgr%report_error("qns_solve", &
318  "No function has been defined.", &
319  nl_invalid_operation_error)
320  return
321  end if
322  if (nvar /= neqn) then
323  ! ERROR: # of equations doesn't match # of variables
324  write(errmsg, '(AI0AI0A)') "The number of equations (", neqn, &
325  ") does not match the number of unknowns (", nvar, ")."
326  call errmgr%report_error("qns_solve", trim(errmsg), &
327  nl_invalid_input_error)
328  return
329  end if
330  flag = 0
331  if (size(x) /= nvar) then
332  flag = 3
333  else if (size(fvec) /= neqn) then
334  flag = 4
335  end if
336  if (flag /= 0) then
337  ! One of the input arrays is not sized correctly
338  write(errmsg, '(AI0A)') "Input number ", flag, &
339  " is not sized correctly."
340  call errmgr%report_error("qns_solve", trim(errmsg), &
341  nl_array_size_error)
342  return
343  end if
344 
345  ! Local Memory Allocation
346  allocate(q(neqn, neqn), stat = flag)
347  if (flag == 0) allocate(r(neqn, nvar), stat = flag)
348  if (flag == 0) allocate(tau(min(neqn, nvar)), stat = flag)
349  if (flag == 0) allocate(b(neqn, nvar), stat = flag)
350  if (flag == 0) allocate(df(neqn), stat = flag)
351  if (flag == 0) allocate(fvold(neqn), stat = flag)
352  if (flag == 0) allocate(xold(nvar), stat = flag)
353  if (flag == 0) allocate(dx(nvar), stat = flag)
354  if (flag == 0) allocate(s(neqn), stat = flag)
355  if (flag /= 0) then
356  ! ERROR: Out of memory
357  call errmgr%report_error("qns_solve", &
358  "Insufficient memory available.", nl_out_of_memory_error)
359  return
360  end if
361  call qr_factor(r, tau, work, lw1)
362  call form_qr(r, tau, q, work, lw2)
363  call fcn%jacobian(x, b, fv = fvec, olwork = lw3)
364  allocate(work(max(lw1, lw2, lw3)), stat = flag)
365  if (flag /= 0) then
366  ! ERROR: Out of memory
367  call errmgr%report_error("qns_solve", &
368  "Insufficient memory available.", nl_out_of_memory_error)
369  return
370  end if
371 
372  ! Test to see if the initial guess is a root
373  call fcn%fcn(x, fvec)
374  f = half * dot_product(fvec, fvec)
375  neval = neval + 1
376  test = zero
377  do i = 1, neqn
378  test = max(abs(fvec(i)), test)
379  end do
380  if (test < ftol) then
381  fcnvrg = .true.
382  end if
383 
384  ! Process
385  flag = 0 ! Used to check for convergence errors
386  if (.not.fcnvrg) then
387  ! Determine the maximum line search step
388  stpmax = factor * max(norm2(x), real(nvar, real64))
389 
390  ! Main Iteration Loop
391  do
392  ! Update the iteration counter
393  iter = iter + 1
394 
395  ! Compute or update the Jacobian
396  if (restart) then
397  ! Compute the Jacobian
398  call fcn%jacobian(x, b, fvec, work)
399  njac = njac + 1
400 
401  ! Compute the QR factorization, and form Q & R
402  r = b ! Copy the Jacobian - we'll need it later
403  call qr_factor(r, tau, work)
404  call form_qr(r, tau, q, work)
405 
406  ! Reset the Jacobian iteration counter
407  jcount = 0
408  else
409  ! Apply the rank 1 update to Q and R
410  df = fvec - fvold
411  dx = x - xold
412  x2 = dot_product(dx, dx)
413 
414  ! Compute S = ALPHA * (DF - B * DX)
415  s = (df - matmul(b, dx))
416  call recip_mult_array(x2, s)
417 
418  ! Compute the new Q and R matrices for the rank1 update:
419  ! B' = B + ALPHA * S * DX**T
420  call rank1_update(one, s, dx, b)
421  call qr_rank1_update(q, r, s, dx, work) ! S & DX overwritten
422 
423  ! Increment the counter tracking how many iterations have
424  ! passed since the last Jacobian recalculation
425  jcount = jcount + 1
426  end if
427 
428  ! Compute GRAD = B**T * F, store in DX
429  call mtx_mult(.true., one, b, fvec, zero, dx)
430 
431  ! Store FVEC and X
432  xold = x
433  fvold = fvec
434  fold = f
435 
436  ! Solve the linear system: B * DX = -F for DX noting that
437  ! B = Q * R. As such, form -Q**T * F, and store in DF
438  call mtx_mult(.true., -one, q, fvec, zero, df)
439 
440  ! Now we have R * DX = -Q**T * F, and since R is upper
441  ! triangular, the solution is readily computed. The solution
442  ! will be stored in the first NVAR elements of DF
443  call solve_triangular_system(.true., .false., .true., r, &
444  df(1:nvar))
445 
446  ! Ensure the new solution estimate is heading in a sensible
447  ! direction. If not, it is likely time to update the Jacobian
448  temp = dot_product(dx, df(1:nvar))
449  if (temp >= zero) then
450  restart = .true.
451  if (this%get_print_status()) then
452  call print_status(iter, neval, njac, xnorm, fnorm)
453  end if
454  cycle
455  end if
456 
457  ! Apply the line search if needed
458  if (this%get_use_line_search()) then
459  ! Define the step length for the line search
460  temp = dot_product(df(1:nvar), df(1:nvar))
461  if (temp > stpmax) df(1:nvar) = df(1:nvar) * (stpmax / temp)
462 
463  ! Apply the line search
464  call limit_search_vector(df(1:nvar), stpmax)
465  call ls%search(fcn, xold, dx, df(1:nvar), x, fvec, fold, &
466  f, lib, errmgr)
467  neval = neval + lib%fcn_count
468  else
469  ! No line search - just update the solution estimate
470  x = x + df(1:nvar)
471  call fcn%fcn(x, fvec)
472  f = half * dot_product(fvec, fvec)
473  neval = neval + 1
474  end if
475 
476  ! Test for convergence
477  if (lib%converge_on_zero_diff .and. &
478  this%get_use_line_search()) then
479  call test_convergence(x, xold, fvec, dx, .true., xtol, &
480  ftol, gtol, check, xcnvrg, fcnvrg, gcnvrg, xnorm, fnorm)
481  else
482  call test_convergence(x, xold, fvec, dx, .false., xtol, &
483  ftol, gtol, check, xcnvrg, fcnvrg, gcnvrg, xnorm, fnorm)
484  end if
485  if (.not.check) then
486  ! The solution did not converge, figure out why
487  if (gcnvrg) then
488  ! The slope of the gradient is sufficiently close to
489  ! zero to cause issue.
490  if (restart) then
491  ! We've already tried recalculating a new Jacobian,
492  ! issue a warning
493  write(errmsg, '(AI0AE8.3AE8.3)') &
494  "It appears the solution has settled to " // &
495  "a point where the slope of the gradient " // &
496  "is effectively zero. " // new_line('c') // &
497  "Function evaluations performed: ", neval, &
498  "." // new_line('c') // &
499  "Change in Variable: ", xnorm, &
500  new_line('c') // "Residual: ", fnorm
501  call errmgr%report_warning("nqs_solve", &
502  trim(errmsg), nl_spurious_convergence_error)
503  exit
504  else
505  ! Try computing a new Jacobian
506  restart = .true.
507  end if
508  else
509  ! We have not converged, but we're not stuck with a
510  ! zero slope gradient vector either. Go ahead and
511  ! continue the iteration process without recomputing
512  ! the Jacobian - unless the user dictates a
513  ! recaclulation.
514  if (jcount >= this%m_jDelta) then
515  restart = .true.
516  else
517  restart = .false.
518  end if
519  end if
520  else
521  ! The solution has converged. It's OK to exit
522  exit
523  end if
524 
525  ! Print status
526  if (this%get_print_status()) then
527  call print_status(iter, neval, njac, xnorm, fnorm)
528  end if
529 
530  ! Ensure we haven't made too many function evaluations
531  if (neval >= maxeval) then
532  flag = 1
533  exit
534  end if
535  end do
536  end if
537 
538  ! Report out iteration statistics
539  if (present(ib)) then
540  ib%iter_count = iter
541  ib%fcn_count = neval
542  ib%jacobian_count = njac
543  ib%gradient_count = 0
544  ib%converge_on_fcn = fcnvrg
545  ib%converge_on_chng = xcnvrg
546  ib%converge_on_zero_diff = gcnvrg
547  end if
548 
549  ! Check for convergence issues
550  if (flag /= 0) then
551  write(errmsg, '(AI0AE8.3AE8.3)') "The algorithm failed to " // &
552  "converge. Function evaluations performed: ", neval, &
553  "." // new_line('c') // "Change in Variable: ", xnorm, &
554  new_line('c') // "Residual: ", fnorm
555  call errmgr%report_error("qns_solve", trim(errmsg), &
556  nl_convergence_error)
557  end if
558  end subroutine

672  subroutine ns_solve(this, fcn, x, fvec, ib, err)
673  ! Arguments
674  class(newton_solver), intent(inout) :: this
675  class(vecfcn_helper), intent(in) :: fcn
676  real(real64), intent(inout), dimension(:) :: x
677  real(real64), intent(out), dimension(:) :: fvec
678  type(iteration_behavior), optional :: ib
679  class(errors), intent(inout), optional, target :: err
680 
681  ! Parameters
682  real(real64), parameter :: zero = 0.0d0
683  real(real64), parameter :: half = 0.5d0
684  real(real64), parameter :: one = 1.0d0
685  real(real64), parameter :: mintol = 1.0d-12
686  real(real64), parameter :: factor = 1.0d2
687 
688  ! Local Variables
689  logical :: check, xcnvrg, fcnvrg, gcnvrg
690  integer(int32) :: i, neqn, nvar, lwork, flag, neval, iter, maxeval, njac
691  integer(int32), allocatable, dimension(:) :: ipvt
692  real(real64), allocatable, dimension(:) :: dir, grad, xold, work
693  real(real64), allocatable, dimension(:,:) :: jac
694  real(real64) :: ftol, xtol, gtol, f, fold, stpmax, xnorm, fnorm, temp, test
695  type(iteration_behavior) :: lib
696  class(errors), pointer :: errmgr
697  type(errors), target :: deferr
698  character(len = 256) :: errmsg
699  class(line_search), allocatable :: ls
700 
701  ! Initialization
702  xcnvrg = .false.
703  fcnvrg = .false.
704  gcnvrg = .false.
705  neqn = fcn%get_equation_count()
706  nvar = fcn%get_variable_count()
707  neval = 0
708  iter = 0
709  njac = 0
710  ftol = this%get_fcn_tolerance()
711  xtol = this%get_var_tolerance()
712  gtol = this%get_gradient_tolerance()
713  maxeval = this%get_max_fcn_evals()
714  if (present(ib)) then
715  ib%iter_count = iter
716  ib%fcn_count = neval
717  ib%jacobian_count = njac
718  ib%gradient_count = 0
719  ib%converge_on_fcn = fcnvrg
720  ib%converge_on_chng = xcnvrg
721  ib%converge_on_zero_diff = gcnvrg
722  end if
723  if (present(err)) then
724  errmgr => err
725  else
726  errmgr => deferr
727  end if
728  if (this%get_use_line_search()) then
729  if (.not.this%is_line_search_defined()) &
730  call this%set_default_line_search()
731  call this%get_line_search(ls)
732  end if
733 
734  ! Input Checking
735  if (.not.fcn%is_fcn_defined()) then
736  ! ERROR: No function is defined
737  call errmgr%report_error("ns_solve", &
738  "No function has been defined.", &
739  nl_invalid_operation_error)
740  return
741  end if
742  if (nvar /= neqn) then
743  ! ERROR: # of equations doesn't match # of variables
744  write(errmsg, '(AI0AI0A)') "The number of equations (", neqn, &
745  ") does not match the number of unknowns (", nvar, ")."
746  call errmgr%report_error("ns_solve", trim(errmsg), &
747  nl_invalid_input_error)
748  return
749  end if
750  flag = 0
751  if (size(x) /= nvar) then
752  flag = 3
753  else if (size(fvec) /= neqn) then
754  flag = 4
755  end if
756  if (flag /= 0) then
757  ! One of the input arrays is not sized correctly
758  write(errmsg, '(AI0A)') "Input number ", flag, &
759  " is not sized correctly."
760  call errmgr%report_error("ns_solve", trim(errmsg), &
761  nl_array_size_error)
762  return
763  end if
764 
765  ! Local Memory Allocation
766  allocate(ipvt(nvar), stat = flag)
767  if (flag == 0) allocate(dir(nvar), stat = flag)
768  if (flag == 0) allocate(grad(nvar), stat = flag)
769  if (flag == 0) allocate(xold(nvar), stat = flag)
770  if (flag == 0) allocate(jac(nvar, neqn), stat = flag)
771  if (flag == 0) then
772  call fcn%jacobian(x, jac, fv = fvec, olwork = lwork)
773  allocate(work(lwork), stat = flag)
774  end if
775  if (flag /= 0) then
776  ! ERROR: Out of memory
777  call errmgr%report_error("ns_solve", &
778  "Insufficient memory available.", nl_out_of_memory_error)
779  return
780  end if
781 
782  ! Test to see if the initial guess is a root
783  call fcn%fcn(x, fvec)
784  f = half * dot_product(fvec, fvec)
785  neval = neval + 1
786  test = zero
787  do i = 1, neqn
788  test = max(abs(fvec(i)), test)
789  end do
790  if (test < ftol) then
791  fcnvrg = .true.
792  end if
793 
794  ! Process
795  flag = 0 ! Used to check for convergence errors
796  if (.not.fcnvrg) then
797  ! Compute the maximum step size for the line search process
798  stpmax = factor * max(norm2(x), real(nvar, real64))
799 
800  ! Main Iteration Loop
801  do
802  ! Increment the iteration counter
803  iter = iter + 1
804 
805  ! Compute the Jacobian
806  call fcn%jacobian(x, jac, fvec, work)
807  njac = njac + 1
808 
809  ! Compute the gradient
810  do i = 1, nvar
811  grad(i) = dot_product(jac(:,i), fvec)
812  end do
813 
814  ! Compute the LU factorization of the Jacobian
815  call lu_factor(jac, ipvt, errmgr)
816  if (errmgr%has_warning_occurred()) then
817  ! The Jacobian is singular - warning was issued already, so
818  ! simply exit the routine. Do not return as a return at
819  ! this point would not allow for proper updating of the
820  ! iteration tracking parameters
821  exit
822  end if
823 
824  ! Store previous iteration values
825  xold = x
826  fold = f
827 
828  ! Define the right-hand-side for the linear system
829  dir = -fvec
830 
831  ! Solve the linear system of equations
832  call solve_lu(jac, ipvt, dir)
833 
834  ! Apply the line search if needed
835  if (this%get_use_line_search()) then
836  ! Define the step length for the line search
837  temp = dot_product(dir, dir)
838  if (temp > stpmax) dir = dir * (stpmax / temp)
839 
840  ! Apply the line search
841  call limit_search_vector(dir, stpmax)
842  call ls%search(fcn, xold, grad, dir, x, fvec, &
843  fold, f, lib, errmgr)
844  neval = neval + lib%fcn_count
845  else
846  ! No line search - just update the solution estimate
847  x = x + dir
848  call fcn%fcn(x, fvec)
849  f = half * dot_product(fvec, fvec)
850  neval = neval + 1
851  end if
852 
853  ! Check for convergence
854  call test_convergence(x, xold, fvec, grad, .true., xtol, &
855  ftol, gtol, check, xcnvrg, fcnvrg, gcnvrg, xnorm, fnorm)
856  if (check) then
857  ! The solution has converged
858  exit
859  else if (gcnvrg) then
860  ! The solution appears to have settled at a point where

987  subroutine brent_solve(this, fcn, x, lim, f, ib, err)

989  class(brent_solver), intent(inout) :: this
990  class(fcn1var_helper), intent(in) :: fcn
991  real(real64), intent(inout) :: x

993  real(real64), intent(out), optional :: f
994  type(iteration_behavior), optional :: ib
995  class(errors), intent(inout), optional, target :: err

997  ! Parameters
998  real(real64), parameter :: zero = 0.0d0
999  real(real64), parameter :: half = 0.5d0
1000  real(real64), parameter :: one = 1.0d0
1001  real(real64), parameter :: two = 2.0d0
1002  real(real64), parameter :: three = 3.0d0
1003 
1004  ! Local Variables
1005  logical :: fcnvrg, xcnvrg
1006  integer(int32) :: neval, maxeval, flag, iter
1007  real(real64) :: ftol, xtol, a, b, c, fa, fb, fc, p, q, r, s, xm, e, d, &
1008  mn1, mn2, eps, tol1, temp
1009  class(errors), pointer :: errmgr
1010  type(errors), target :: deferr
1011  character(len = 256) :: errmsg
1012 
1013  ! Initialization
1014  fcnvrg = .false.
1015  xcnvrg = .false.
1016  x = zero
1017  a = min(lim%x1, lim%x2)
1018  b = max(lim%x1, lim%x2)
1019  neval = 0
1020  iter = 0
1021  eps = epsilon(eps)
1022  ftol = this%get_fcn_tolerance()
1023  xtol = this%get_var_tolerance()
1024  maxeval = this%get_max_fcn_evals()
1025  if (present(f)) f = zero
1026  if (present(ib)) then
1027  ib%iter_count = iter
1028  ib%fcn_count = neval
1029  ib%jacobian_count = 0
1030  ib%gradient_count = 0
1031  ib%converge_on_fcn = fcnvrg
1032  ib%converge_on_chng = xcnvrg
1033  ib%converge_on_zero_diff = .false.
1034  end if
1035  if (present(err)) then
1036  errmgr => err
1037  else
1038  errmgr => deferr
1039  end if
1040 
1041  ! Input Check
1042  if (.not.fcn%is_fcn_defined()) then
1043  ! ERROR: No function is defined
1044  call errmgr%report_error("brent_solve", &
1045  "No function has been defined.", &
1046  nl_invalid_operation_error)
1047  return
1048  end if
1049  if (abs(a - b) < eps) then
1050  ! ERROR: Search limits are too tight
1051  write(errmsg, '(AE8.3AE8.3)') "Search limits have no " // &
1052  "appreciable difference between them. Lower Limit: ", a, &
1053  ", Upper Limit: ", b
1054  call errmgr%report_error("brent_solve", trim(errmsg), &
1055  nl_invalid_operation_error)
1056  return
1057  end if
1058 
1059  ! Process
1060  flag = 0
1061  fa = fcn%fcn(a)
1062  fb = fcn%fcn(b)
1063  neval = 2
1064  fc = fb
1065  do
1066  ! Increment the iteration counter
1067  iter = iter + 1
1068 
1069  ! Adjust the bounding interval
1070  if ((fb > zero .and. fc >= zero) .or. &
1071  (fb < zero .and. fc < zero)) then
1072  c = a
1073  fc = fa
1074  d = b - a
1075  e = d
1076  end if
1077  if (abs(fc) < abs(fb)) then
1078  a = b
1079  b = c
1080  c = a
1081  fa = fb
1082  fb = fc
1083  fc = fa
1084  end if
1085 
1086  ! Convergence Check
1087  tol1 = two * eps * abs(b) + half * xtol
1088  xm = half * (c - b)
1089  if (abs(fb) < ftol) then
1090  x = b
1091  fcnvrg = .true.
1092  exit
1093  end if
1094  if (abs(xm) <= tol1) then
1095  x = b
1096  xcnvrg = .true.
1097  exit
1098  end if
1099 
1100  ! Actual Method
1101  if (abs(e) >= tol1 .and. abs(fa) > abs(fb)) then
1102  ! Attempt the inverse quadratic interpolation to determine
1103  ! the root
1104  s = fb / fa
1105  if (abs(a - c) < eps) then ! a == c
1106  p = two * xm * s
1107  q = one - s
1108  else
1109  q = fa / fc
1110  r = fb / fc
1111  p = s * (two * xm * q * (q - r) - (b - a) * (r - one))
1112  q = (q - one) * (r - one) * (s - one)
1113  end if
1114 
1115  ! Ensure we're within bounds
1116  if (p > zero) q = -q
1117  p = abs(p)
1118  mn1 = three * xm * q - abs(tol1 * q)
1119  mn2 = abs(e * q)
1120  if (mn1 < mn2) then
1121  temp = mn1
1122  else
1123  temp = mn2
1124  end if
1125  if (two * p < temp) then
1126  ! Accept the interpolation
1127  e = d
1128  d = p / q
1129  else
1130  ! The interpolation failed, use bisection
1131  d = xm
1132  e = d
1133  end if
1134  else
1135  ! The bounds are decreasing too slowly, use bisection
1136  d = xm
1137  e = d
1138  end if
1139 
1140  ! Move the last best guess to the lower limit parameter (A)
1141  a = b
1142  fa = fb
1143  if (abs(d) > tol1) then
1144  b = b + d
1145  else
1146  b = b + sign(tol1, xm)
1147  end if
1148  fb = fcn%fcn(b)
1149  neval = neval + 1
1150 
1151  ! Print iteration status
1152  if (this%get_print_status()) then
1153  call print_status(iter, neval, 0, xm, fb)
1154  end if
1155 
1156  ! Ensure we haven't made too many function evaluations
1157  if (neval >= maxeval) then
1158  flag = 1
1159  exit
1160  end if
1161  end do
1162 
1163  ! Report out iteration statistics and other optional outputs
1164  if (present(f)) f = fb
1165  if (present(ib)) then
1166  ib%iter_count = iter
1167  ib%fcn_count = neval
1168  ib%jacobian_count = 0
1169  ib%gradient_count = 0
1170  ib%converge_on_fcn = fcnvrg
1171  ib%converge_on_chng = xcnvrg
1172  ib%converge_on_zero_diff = .false.
1173  end if
1174 
1175  ! Check for convergence issues
1176  if (flag /= 0) then
1177  write(errmsg, '(AI0AE8.3AE8.3)') "The algorithm failed to " // &
1178  "converge. Function evaluations performed: ", neval, &
1179  "." // new_line('c') // "Change in Variable: ", xm, &
1180  new_line('c') // "Residual: ", fb
1181  call errmgr%report_error("brent_solve", trim(errmsg), &
1182  nl_convergence_error)
1183  end if
