195  subroutine nm_solve(this, fcn, x, fout, ib, err)
196  ! Arguments
197  class(nelder_mead), intent(inout) :: this
198  class(fcnnvar_helper), intent(in) :: fcn
199  real(real64), intent(inout), dimension(:) :: x
200  real(real64), intent(out), optional :: fout
201  type(iteration_behavior), optional :: ib
202  class(errors), intent(inout), optional, target :: err
203 
204  ! Parameters
205  real(real64), parameter :: zero = 0.0d0
206  real(real64), parameter :: negone = -1.0d0
207  real(real64), parameter :: half = 0.5d0
208  real(real64), parameter :: two = 2.0d0
209 
210  ! Local Variables
211  logical :: buildSimplex, fcnvrg
212  integer(int32) :: i, ihi, ilo, ihi2, ndim, npts, flag, neval, iter, &
213  maxeval
214  real(real64) :: ftol, rtol, ftry, fsave, fval, swp
215  real(real64), allocatable, dimension(:) :: f, pcent, pmin, work
216  class(errors), pointer :: errmgr
217  type(errors), target :: deferr
218  character(len = 256) :: errmsg
219 
220  ! Initialization
221  ndim = fcn%get_variable_count()
222  npts = ndim + 1
223  buildsimplex = .true.
224  maxeval = this%get_max_fcn_evals()
225  ftol = this%get_tolerance()
226  iter = 0
227  neval = 0
228  fcnvrg = .false.
229  if (present(ib)) then
230  ib%iter_count = iter
231  ib%fcn_count = neval
232  ib%jacobian_count = 0
233  ib%gradient_count = 0
234  ib%converge_on_fcn = fcnvrg
235  ib%converge_on_chng = .false.
236  ib%converge_on_zero_diff = .false.
237  end if
238  if (present(err)) then
239  errmgr => err
240  else
241  errmgr => deferr
242  end if
243 
244  ! Input Check
245  if (.not.fcn%is_fcn_defined()) then
246  ! ERROR: No function is defined
247  call errmgr%report_error("nm_solve", &
248  "No function has been defined.", &
249  nl_invalid_operation_error)
250  return
251  end if
252  if (size(x) /= ndim) then
253  write(errmsg, '(AI0AI0A)') &
254  "It was expected to receive a coordinate vector of length ", &
255  ndim, " , but a vector of length ", size(x), " was received."
256  call errmgr%report_error("nm_solve", trim(errmsg), &
257  nl_invalid_input_error)
258  return
259  end if
260 
261  ! Ensure that if an initial simplex was defined, that it is
262  ! appropriately sized. If not, simply create a new simplex of the
263  ! appropriate size.
264  if (allocated(this%m_simplex)) then
265  ! This matrix must be NDIM-by-NPTS
266  if (size(this%m_simplex, 1) /= ndim .or. &
267  size(this%m_simplex, 2) /= npts) then
268  deallocate(this%m_simplex)
269  buildsimplex = .true.
270  else
271  ! The simplex is appropriately sized
272  buildsimplex = .false.
273  end if
274  end if
275 
276  ! Local Memory Allocation
277  allocate(f(npts), stat = flag)
278  if (flag == 0) allocate(pcent(ndim), stat = flag)
279  if (flag == 0) allocate(pmin(ndim), stat = flag)
280  if (flag == 0) allocate(work(ndim), stat = flag)
281  if (buildsimplex .and. flag == 0) allocate(this%m_simplex(ndim, npts))
282  if (flag /= 0) then
283  ! ERROR: Out of memory
284  call errmgr%report_error("nm_solve", &
285  "Insufficient memory available.", nl_out_of_memory_error)
286  return
287  end if
288 
289  ! Define the initial simplex, if needed
290  if (buildsimplex) then
291  this%m_simplex(:,1) = x
292  do i = 2, npts
293  this%m_simplex(:,i) = x
294  end do
295  do i = 1, ndim
296  this%m_simplex(i,i+1) = this%m_simplex(i,i+1) + this%m_initSize
297  end do
298  end if
299 
300  ! Evaluate the function at each vertex of the simplex
301  do i = 1, npts
302  f(i) = fcn%fcn(this%m_simplex(:,i))
303  end do
304  neval = npts
305  fval = f(1)
306 
307  do i = 1, ndim
308  pcent(i) = sum(this%m_simplex(i,:))
309  end do
310 
311  ! Main Loop
312  flag = 0 ! Used to check for convergence errors
313  do
314  ! Update the iteration counter
315  iter = iter + 1
316 
317  ! Determine the characteristics of each vertex
318  ilo = 1
319  if (f(1) > f(2)) then
320  ihi = 1
321  ihi2 = 2
322  else
323  ihi = 2
324  ihi2 = 1
325  end if
326  do i = 1, npts
327  if (f(i) <= f(ilo)) ilo = i
328  if (f(i) > f(ihi)) then
329  ihi2 = ihi
330  ihi = i
331  else if (f(i) > f(ihi2)) then
332  if (i /= ihi) ihi2 = i
333  end if
334  end do
335 
336  ! Check for convergence. Nelder and Mead recommend using the
337  ! following convergence test: sqrt(sum(f - favg)**2 / n); however,
338  ! it seems that a sufficient check may be made using only the
339  ! extreme function values of the simplex (highest and lowest valued
340  ! points).
341  rtol = abs(f(ihi) - f(ilo))
342  if (rtol < ftol) then
343  swp = f(1)
344  f(1) = f(ilo)
345  f(ilo) = swp
346  do i = 1, ndim
347  swp = this%m_simplex(i,1)
348  this%m_simplex(i,1) = this%m_simplex(i,ilo)
349  this%m_simplex(i,ilo) = swp
350  x(i) = this%m_simplex(i,1)
351  end do
352  fval = f(1)
353  fcnvrg = .true.
354  exit
355  end if
356 
357  ! Start of a new iteration by reflecting the simplex at its largest
358  ! point.
359  ftry = this%extrapolate(fcn, f, pcent, ihi, negone, neval, work)
360  if (ftry <= f(ilo)) then
361  ! The result of the reflection is better than the current
362  ! best point. As a result, try a factor of 2 in the reflected
363  ! direction. Again, the highest point is of interest.
364  ftry = this%extrapolate(fcn, f, pcent, ihi, two, neval, work)
365  else if (ftry >= f(ihi2)) then
366  ! The reflected point is worse than the second highest, so look
367  ! for an intermediate lower point (contract the simplex)
368  fsave = f(ihi)
369  ftry = this%extrapolate(fcn, f, pcent, ihi, half, neval, work)
370  if (ftry >= fsave) then
371  ! Cannot improve on the high point. Try to contract around
372  ! the low point.
373  do i = 1, npts
374  if (i /= ilo) then
375  pcent = half * (this%m_simplex(:,i) + &
376  this%m_simplex(:,ilo))
377  this%m_simplex(:,i) = pcent
378  f(i) = fcn%fcn(pcent)
379  end if
380  end do
381  neval = neval + npts
382  do i = 1, ndim
383  pcent(i) = sum(this%m_simplex(i,:))
384  end do
385  end if
386  end if
387 
388  ! Print iteration status
389  if (this%get_print_status()) then
390  print *, ""
391  print '(AI0)', "Iteration: ", iter
392  print '(AI0)', "Function Evaluations: ", neval
393  print '(AE8.3)', "Function Value: ", fval
394  print '(AE8.3)', "Convergence Parameter: ", rtol
395  end if
396 
397  ! Ensure we haven't made too many function evaluations
398  if (neval >= maxeval) then
399  flag = 1
400  exit
401  end if
402  end do
403 
404  ! Report out iteration statistics
405  if (present(ib)) then
406  ib%iter_count = iter
407  ib%fcn_count = neval
408  ib%jacobian_count = 0
409  ib%gradient_count = 0
410  ib%converge_on_fcn = fcnvrg
411  ib%converge_on_chng = .false.
412  ib%converge_on_zero_diff = .false.
413  end if
414 
415  ! Get the function value at the computed minimum
416  if (present(fout)) fout = fval
417 
418  ! Check for convergence issues
419  if (flag /= 0) then
420  write(errmsg, '(AI0AE8.3AE8.3)') &
421  "The algorithm failed to converge." // new_line('c') // &
422  "Function evaluations performed: ", neval, new_line('c') // &
423  "Convergence Parameter: ", rtol, new_line('c') // &
424  "Convergence Criteria: ", ftol
425  call errmgr%report_error("nm_solve", trim(errmsg), &
426  nl_convergence_error)
427  end if
428  end subroutine

445  function nm_extrapolate(this, fcn, y, pcent, ihi, fac, neval, work) &
446  result(ytry)
447  ! Arguments
448  class(nelder_mead), intent(inout) :: this
449  class(fcnnvar_helper), intent(in) :: fcn
450  real(real64), intent(inout), dimension(:) :: y, pcent
451  integer(int32), intent(in) :: ihi
452  real(real64), intent(in) :: fac
453  integer(int32), intent(inout) :: neval
454  real(real64), intent(out), dimension(:) :: work
455  real(real64) :: ytry
456 
457  ! Parameters
458  real(real64), parameter :: one = 1.0d0
459 
460  ! Local Variables
461  integer(int32) :: i, ndim
462  real(real64) :: fac1, fac2
463 
464  ! Initialization
465  ndim = size(this%m_simplex, 1)
466 
467  ! Define a trial point
468  fac1 = (one - fac) / ndim
469  fac2 = fac1 - fac
470  do i = 1, ndim
471  work(i) = pcent(i) * fac1 - this%m_simplex(i,ihi) * fac2
472  end do
473 
474  ! Evaluate the function at the trial point, and then replace if the
475  ! trial provides an improvement
476  ytry = fcn%fcn(work)
477  neval = neval + 1
478  if (ytry < y(ihi)) then
479  y(ihi) = ytry
480  do i = 1, ndim
481  pcent(i) = pcent(i) + work(i) - this%m_simplex(i,ihi)
482  this%m_simplex(i,ihi) = work(i)
483  end do
484  end if
485  end function

730  subroutine bfgs_solve(this, fcn, x, fout, ib, err)
731  ! Arguments
732  class(bfgs), intent(inout) :: this
733  class(fcnnvar_helper), intent(in) :: fcn
734  real(real64), intent(inout), dimension(:) :: x
735  real(real64), intent(out), optional :: fout
736  type(iteration_behavior), optional :: ib
737  class(errors), intent(inout), optional, target :: err
738 
739  ! Parameters
740  real(real64), parameter :: zero = 0.0d0
741  real(real64), parameter :: one = 1.0d0
742  real(real64), parameter :: negone = -1.0d0
743  real(real64), parameter :: factor = 1.0d2
744  real(real64), parameter :: small = 1.0d-10
745 
746  ! Local Variables
747  logical :: xcnvrg, gcnvrg
748  integer(int32) :: i, n, maxeval, neval, ngrad, flag, iter
749  real(real64) :: xtol, gtol, fp, stpmax, fret, xtest, gtest, temp, ydx
750  real(real64), allocatable, dimension(:) :: g, dx, u, v, y, gold, xnew, bdx
751  real(real64), allocatable, dimension(:,:) :: b, r
752  class(errors), pointer :: errmgr
753  type(errors), target :: deferr
754  character(len = 256) :: errmsg
755  type(iteration_behavior) :: lib
756  class(line_search), allocatable :: ls
757 
758  ! Initialization
759  n = fcn%get_variable_count()
760  maxeval = this%get_max_fcn_evals()
761  gtol = this%get_tolerance()
762  xtol = this%get_var_tolerance()
763  iter = 0
764  neval = 0
765  ngrad = 0
766  xcnvrg = .false.
767  gcnvrg = .false.
768  if (present(ib)) then
769  ib%iter_count = iter
770  ib%fcn_count = neval
771  ib%jacobian_count = 0
772  ib%gradient_count = ngrad
773  ib%converge_on_fcn = .false.
774  ib%converge_on_chng = xcnvrg
775  ib%converge_on_zero_diff = gcnvrg
776  end if
777  if (present(err)) then
778  errmgr => err
779  else
780  errmgr => deferr
781  end if
782  if (this%get_use_line_search()) then
783  if (.not.this%is_line_search_defined()) &
784  call this%set_default_line_search()
785  call this%get_line_search(ls)
786  end if
787 
788  ! Input Check
789  if (.not.fcn%is_fcn_defined()) then
790  ! ERROR: No function is defined
791  call errmgr%report_error("bfgs_solve", &
792  "No function has been defined.", &
793  nl_invalid_operation_error)
794  return
795  end if
796  if (size(x) /= n) then
797  write(errmsg, '(AI0AI0A)') &
798  "It was expected to receive a coordinate vector of length ", &
799  n, " , but a vector of length ", size(x), " was received."
800  call errmgr%report_error("bfgs_solve", trim(errmsg), &
801  nl_invalid_input_error)
802  return
803  end if
804 
805  ! Local Memory Allocation
806  allocate(g(n), stat = flag)
807  if (flag == 0) allocate(dx(n), stat = flag)
808  if (flag == 0) allocate(u(n), stat = flag)
809  if (flag == 0) allocate(v(n), stat = flag)
810  if (flag == 0) allocate(y(n), stat = flag)
811  if (flag == 0) allocate(bdx(n), stat = flag)
812  if (flag == 0) allocate(gold(n), stat = flag)
813  if (flag == 0) allocate(xnew(n), stat = flag)
814  if (flag == 0) allocate(b(n,n), stat = flag)
815  if (flag == 0) allocate(r(n,n), stat = flag)
816  if (flag /= 0) then
817  ! ERROR: Memory Error
818  call errmgr%report_error("bfgs_solve", &
819  "Insufficient memory available.", nl_out_of_memory_error)
820  return
821  end if
822 
823  ! Process
824  fp = fcn%fcn(x)
825  call fcn%gradient(x, g, fp)
826  neval = 1
827  ngrad = 1
828 
829  ! Check for a "zero" gradient at the initial point
830  gtest = norm2(g)
831  if (gtest < gtol) then
832  gcnvrg = .true.
833  end if
834 
835  ! Main Loop
836  flag = 0
837  if (.not.gcnvrg) then
838  do
839  ! Update the iteration counter
840  iter = iter + 1
841 
842  ! Define the initial direction, and a limit on the line search
843  ! step
844  if (iter == 1) then
845  dx = -g
846  stpmax = factor * max(norm2(x), real(n, real64))
847  end if
848 
849  ! Perform the line search
850  if (this%get_use_line_search()) then
851  call limit_search_vector(dx, stpmax)
852  call ls%search(fcn, x, g, dx, xnew, fp, fret, lib, errmgr)
853  neval = neval + lib%fcn_count
854  fp = fret
855  else
856  xnew = x + dx
857  fp = fcn%fcn(xnew)
858  neval = neval + 1
859  end if
860 
861  ! Update the gradient and line direction
862  do i = 1, n
863  dx(i) = xnew(i) - x(i)
864  x(i) = xnew(i)
865  gold(i) = g(i)
866  end do
867  call fcn%gradient(x, g, fp)
868  ngrad = ngrad + 1
869 
870  ! Test for convergence on the change in X
871  xtest = zero
872  do i = 1, n
873  temp = abs(dx(i)) / max(abs(x(i)), one)
874  xtest = max(temp, xtest)
875  end do
876  if (xtest < xtol) then
877  xcnvrg = .true.
878  exit
879  end if
880 
881  ! Test for convergence on the gradient
882  gtest = norm2(g)
883  if (gtest < gtol) then
884  gcnvrg = .true.
885  exit
886  end if
887 
888  ! Perform the BFGS update
889  y = g - gold
890  ydx = dot_product(y, dx)
891 
892  ! Establish an initial approximation to the Hessian matrix
893  if (iter == 1) then
894  temp = sqrt(dot_product(y, y) / ydx)
895  call dlaset('A', n, n, zero, temp, r, n)
896  end if
897 
898  ! Compute: B = R**T * R
899  call tri_mtx_mult(.true., one, r, zero, b)
900 
901  ! Compute bdx = B * dX (B is symmetric)
902  call dsymv('u', n, one, b, n, dx, 1, zero, bdx, 1)
903 
904  ! Perform the actual update
