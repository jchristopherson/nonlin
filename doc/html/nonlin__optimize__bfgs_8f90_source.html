5  module subroutine bfgs_solve(this, fcn, x, fout, ib, err)
6  ! Arguments
7  class(bfgs), intent(inout) :: this
8  class(fcnnvar_helper), intent(in) :: fcn
9  real(real64), intent(inout), dimension(:) :: x
10  real(real64), intent(out), optional :: fout
11  type(iteration_behavior), optional :: ib
12  class(errors), intent(inout), optional, target :: err
13 
14  ! Parameters
15  real(real64), parameter :: zero = 0.0d0
16  real(real64), parameter :: one = 1.0d0
17  real(real64), parameter :: negone = -1.0d0
18  real(real64), parameter :: factor = 1.0d2
19  real(real64), parameter :: small = 1.0d-10
20 
21  ! Local Variables
22  logical :: xcnvrg, gcnvrg
23  integer(int32) :: i, n, maxeval, neval, ngrad, flag, iter
24  real(real64) :: xtol, gtol, fp, stpmax, fret, xtest, gtest, temp, ydx
25  real(real64), allocatable, dimension(:) :: g, dx, u, v, y, gold, xnew, bdx
26  real(real64), allocatable, dimension(:,:) :: b, r
27  class(errors), pointer :: errmgr
28  type(errors), target :: deferr
29  character(len = 256) :: errmsg
30  type(iteration_behavior) :: lib
31  class(line_search), allocatable :: ls
32 
33  ! Initialization
34  n = fcn%get_variable_count()
35  maxeval = this%get_max_fcn_evals()
36  gtol = this%get_tolerance()
37  xtol = this%get_var_tolerance()
38  iter = 0
39  neval = 0
40  ngrad = 0
41  xcnvrg = .false.
42  gcnvrg = .false.
43  if (present(ib)) then
44  ib%iter_count = iter
45  ib%fcn_count = neval
46  ib%jacobian_count = 0
47  ib%gradient_count = ngrad
48  ib%converge_on_fcn = .false.
49  ib%converge_on_chng = xcnvrg
50  ib%converge_on_zero_diff = gcnvrg
51  end if
52  if (present(err)) then
53  errmgr => err
54  else
55  errmgr => deferr
56  end if
57  if (this%get_use_line_search()) then
58  if (.not.this%is_line_search_defined()) &
59  call this%set_default_line_search()
60  call this%get_line_search(ls)
61  end if
62 
63  ! Input Check
64  if (.not.fcn%is_fcn_defined()) then
65  ! ERROR: No function is defined
66  call errmgr%report_error("bfgs_solve", &
67  "No function has been defined.", &
68  nl_invalid_operation_error)
69  return
70  end if
71  if (size(x) /= n) then
72  write(errmsg, '(AI0AI0A)') &
73  "It was expected to receive a coordinate vector of length ", &
74  n, " , but a vector of length ", size(x), " was received."
75  call errmgr%report_error("bfgs_solve", trim(errmsg), &
76  nl_invalid_input_error)
77  return
78  end if
79 
80  ! Local Memory Allocation
81  allocate(g(n), stat = flag)
82  if (flag == 0) allocate(dx(n), stat = flag)
83  if (flag == 0) allocate(u(n), stat = flag)
84  if (flag == 0) allocate(v(n), stat = flag)
85  if (flag == 0) allocate(y(n), stat = flag)
86  if (flag == 0) allocate(bdx(n), stat = flag)
87  if (flag == 0) allocate(gold(n), stat = flag)
88  if (flag == 0) allocate(xnew(n), stat = flag)
89  if (flag == 0) allocate(b(n,n), stat = flag)
90  if (flag == 0) allocate(r(n,n), stat = flag)
91  if (flag /= 0) then
92  ! ERROR: Memory Error
93  call errmgr%report_error("bfgs_solve", &
94  "Insufficient memory available.", nl_out_of_memory_error)
95  return
96  end if
97 
98  ! Process
99  fp = fcn%fcn(x)
100  call fcn%gradient(x, g, fp)
101  neval = 1
102  ngrad = 1
103 
104  ! Check for a "zero" gradient at the initial point
105  gtest = norm2(g)
106  if (gtest < gtol) then
107  gcnvrg = .true.
108  end if
109 
110  ! Main Loop
111  flag = 0
112  if (.not.gcnvrg) then
113  do
114  ! Update the iteration counter
115  iter = iter + 1
116 
117  ! Define the initial direction, and a limit on the line search
118  ! step
119  if (iter == 1) then
120  dx = -g
121  stpmax = factor * max(norm2(x), real(n, real64))
122  end if
123 
124  ! Perform the line search
125  if (this%get_use_line_search()) then
126  call limit_search_vector(dx, stpmax)
127  call ls%search(fcn, x, g, dx, xnew, fp, fret, lib, errmgr)
128  neval = neval + lib%fcn_count
129  fp = fret
130  else
131  xnew = x + dx
132  fp = fcn%fcn(xnew)
133  neval = neval + 1
134  end if
135 
136  ! Update the gradient and line direction
137  do i = 1, n
138  dx(i) = xnew(i) - x(i)
139  x(i) = xnew(i)
140  gold(i) = g(i)
141  end do
142  call fcn%gradient(x, g, fp)
143  ngrad = ngrad + 1
144 
145  ! Test for convergence on the change in X
146  xtest = zero
147  do i = 1, n
148  temp = abs(dx(i)) / max(abs(x(i)), one)
149  xtest = max(temp, xtest)
150  end do
151  if (xtest < xtol) then
152  xcnvrg = .true.
153  exit
154  end if
155 
156  ! Test for convergence on the gradient
157  gtest = norm2(g)
158  if (gtest < gtol) then
159  gcnvrg = .true.
160  exit
161  end if
162 
163  ! Perform the BFGS update
164  y = g - gold
165  ydx = dot_product(y, dx)
166 
167  ! Establish an initial approximation to the Hessian matrix
168  if (iter == 1) then
169  temp = sqrt(dot_product(y, y) / ydx)
170  call dlaset('A', n, n, zero, temp, r, n)
171  end if
172 
173  ! Compute: B = R**T * R
174  call tri_mtx_mult(.true., one, r, zero, b)
175 
176  ! Compute bdx = B * dX (B is symmetric)
177  call dsymv('u', n, one, b, n, dx, 1, zero, bdx, 1)
178 
179  ! Perform the actual update
180  if (ydx > small) then
181  ! Compute the rank 1 update and downdate
182  u = y / sqrt(ydx)
183  v = bdx / sqrt(dot_product(dx, bdx))
184  call cholesky_rank1_update(r, u)
185  call cholesky_rank1_downdate(r, v)
186  end if ! Else just skip the update
187 
188  ! Compute the solution to: B * dx = -g = (R**T * R) * dx
189  dx = -g
190  call solve_cholesky(.true., r, dx)
191 
192  ! Print iteration status
193  if (this%get_print_status()) then
194  print *, ""
195  print '(AI0)', "Iteration: ", iter
196  print '(AI0)', "Function Evaluations: ", neval
197  print '(AE8.3)', "Function Value: ", fp
198  print '(AE8.3)', "Change in Variable: ", xtest
199  print '(AE8.3)', "Gradient: ", gtest
200  end if
201 
202  ! Ensure we haven't made too many function evaluations
203  if (neval >= maxeval) then
204  flag = 1
205  exit
206  end if
207  end do
208  end if
209 
210  ! Report out iteration statistics
211  if (present(ib)) then
212  ib%iter_count = iter
213  ib%fcn_count = neval
214  ib%jacobian_count = 0
215  ib%gradient_count = ngrad
216  ib%converge_on_fcn = .false.
217  ib%converge_on_chng = xcnvrg
218  ib%converge_on_zero_diff = gcnvrg
219  end if
220 
221  ! Get the function value at the computed minimum
222  if (present(fout)) fout = fp
223 
224  ! Check for convergence issues
225  if (flag /= 0) then
226  write(errmsg, '(AI0AE8.3AE8.3AE8.3)') &
227  "The algorithm failed to converge." // new_line('c') // &
228  "Function evaluations performed: ", neval, new_line('c') // &
229  "Function Value: ", fp, new_line('c') // &
230  "Change in Variable: ", xtest, new_line('c') // &
231  "Gradient: ", gtest
232  call errmgr%report_error("bfgs_solve", trim(errmsg), &
233  nl_convergence_error)
234  end if
235  end subroutine
236 end submodule
