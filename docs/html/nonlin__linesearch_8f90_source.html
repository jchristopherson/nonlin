620  pure function min_backtrack_search(mode, f0, f, f1, alam, alam1, &
621  slope) result(lam)
622  ! Arguments
623  integer(int32), intent(in) :: mode
624  real(real64), intent(in) :: f0, f, f1, alam, alam1, slope
625  real(real64) :: lam
626 
627  ! Parameters
628  real(real64), parameter :: zero = 0.0d0
629  real(real64), parameter :: p5 = 0.5d0
630  real(real64), parameter :: two = 2.0d0
631  real(real64), parameter :: three = 3.0d0
632 
633  ! Local Variables
634  real(real64) :: rhs1, rhs2, a, b, disc
635 
636  ! Process
637  if (mode == 1) then
638  ! Use a quadratic function approximation
639  lam = -slope / (two * (f - f0 - slope))
640  else
641  ! Use a cubic function
642  rhs1 = f - f0 - alam * slope
643  rhs2 = f1 - f0 - alam1 * slope
644  a = (rhs1 / alam**2 - rhs2 / alam1**2) / (alam - alam1)
645  b = (-alam1 * rhs1 / alam**2 + alam * rhs2 / alam1**2) / &
646  (alam - alam1)
647  if (a == zero) then
648  lam = -slope / (two * b)
649  else
650  disc = b**2 - three * a * slope
651  if (disc < zero) then
652  lam = p5 * alam
653  else if (b <= zero) then
654  lam = (-b + sqrt(disc)) / (three * a)
655  else
656  lam = -slope / (b + sqrt(disc))
657  end if
658  end if
659  if (lam > p5 * alam) lam = p5 * alam
660  end if
661  end function
