5! ------------------------------------------------------------------------------
6 module subroutine ns_solve(this, fcn, x, fvec, ib, err)
7 ! Arguments
8 class(newton_solver), intent(inout) :: this
9 class(vecfcn_helper), intent(in) :: fcn
10 real(real64), intent(inout), dimension(:) :: x
11 real(real64), intent(out), dimension(:) :: fvec
12 type(iteration_behavior), optional :: ib
13 class(errors), intent(inout), optional, target :: err
14
15 ! Parameters
16 real(real64), parameter :: zero = 0.0d0
17 real(real64), parameter :: half = 0.5d0
18 real(real64), parameter :: one = 1.0d0
19 real(real64), parameter :: mintol = 1.0d-12
20 real(real64), parameter :: factor = 1.0d2
21
22 ! Local Variables
23 logical :: check, xcnvrg, fcnvrg, gcnvrg
24 integer(int32) :: i, neqn, nvar, lwork, flag, neval, iter, maxeval, njac
25 integer(int32), allocatable, dimension(:) :: ipvt
26 real(real64), allocatable, dimension(:) :: dir, grad, xold, work
27 real(real64), allocatable, dimension(:,:) :: jac
28 real(real64) :: ftol, xtol, gtol, f, fold, stpmax, xnorm, fnorm, temp, test
29 type(iteration_behavior) :: lib
30 class(errors), pointer :: errmgr
31 type(errors), target :: deferr
32 character(len = 256) :: errmsg
33 class(line_search), allocatable :: ls
34
35 ! Initialization
36 xcnvrg = .false.
37 fcnvrg = .false.
38 gcnvrg = .false.
39 neqn = fcn%get_equation_count()
40 nvar = fcn%get_variable_count()
41 neval = 0
42 iter = 0
43 njac = 0
44 ftol = this%get_fcn_tolerance()
45 xtol = this%get_var_tolerance()
46 gtol = this%get_gradient_tolerance()
47 maxeval = this%get_max_fcn_evals()
48 if (present(ib)) then
49 ib%iter_count = iter
50 ib%fcn_count = neval
51 ib%jacobian_count = njac
52 ib%gradient_count = 0
53 ib%converge_on_fcn = fcnvrg
54 ib%converge_on_chng = xcnvrg
55 ib%converge_on_zero_diff = gcnvrg
56 end if
57 if (present(err)) then
58 errmgr => err
59 else
60 errmgr => deferr
61 end if
62 if (this%get_use_line_search()) then
63 if (.not.this%is_line_search_defined()) &
64 call this%set_default_line_search()
65 call this%get_line_search(ls)
66 end if
67
68 ! Input Checking
69 if (.not.fcn%is_fcn_defined()) then
70 ! ERROR: No function is defined
71 call errmgr%report_error("ns_solve", &
72 "No function has been defined.", &
73 nl_invalid_operation_error)
74 return
75 end if
76 if (nvar /= neqn) then
77 ! ERROR: # of equations doesn't match # of variables
78 write(errmsg, '(AI0AI0A)') "The number of equations (", neqn, &
79 ") does not match the number of unknowns (", nvar, ")."
80 call errmgr%report_error("ns_solve", trim(errmsg), &
81 nl_invalid_input_error)
82 return
83 end if
84 flag = 0
85 if (size(x) /= nvar) then
86 flag = 3
87 else if (size(fvec) /= neqn) then
88 flag = 4
89 end if
90 if (flag /= 0) then
91 ! One of the input arrays is not sized correctly
92 write(errmsg, '(AI0A)') "Input number ", flag, &
93 " is not sized correctly."
94 call errmgr%report_error("ns_solve", trim(errmsg), &
95 nl_array_size_error)
96 return
97 end if
98
99 ! Local Memory Allocation
100 allocate(ipvt(nvar), stat = flag)
101 if (flag == 0) allocate(dir(nvar), stat = flag)
102 if (flag == 0) allocate(grad(nvar), stat = flag)
103 if (flag == 0) allocate(xold(nvar), stat = flag)
104 if (flag == 0) allocate(jac(nvar, neqn), stat = flag)
105 if (flag == 0) then
106 call fcn%jacobian(x, jac, fv = fvec, olwork = lwork)
107 allocate(work(lwork), stat = flag)
108 end if
109 if (flag /= 0) then
110 ! ERROR: Out of memory
111 call errmgr%report_error("ns_solve", &
112 "Insufficient memory available.", nl_out_of_memory_error)
113 return
114 end if
115
116 ! Test to see if the initial guess is a root
117 call fcn%fcn(x, fvec)
118 f = half * dot_product(fvec, fvec)
119 neval = neval + 1
120 test = zero
121 do i = 1, neqn
122 test = max(abs(fvec(i)), test)
123 end do
124 if (test < ftol) then
125 fcnvrg = .true.
126 end if
127
128 ! Process
129 flag = 0 ! Used to check for convergence errors
130 if (.not.fcnvrg) then
131 ! Compute the maximum step size for the line search process
132 stpmax = factor * max(norm2(x), real(nvar, real64))
133
134 ! Main Iteration Loop
135 do
136 ! Increment the iteration counter
137 iter = iter + 1
138
139 ! Compute the Jacobian
140 call fcn%jacobian(x, jac, fvec, work)
141 njac = njac + 1
142
143 ! Compute the gradient
144 do i = 1, nvar
145 grad(i) = dot_product(jac(:,i), fvec)
146 end do
147
148 ! Compute the LU factorization of the Jacobian
149 call lu_factor(jac, ipvt, errmgr)
150 if (errmgr%has_warning_occurred()) then
151 ! The Jacobian is singular - warning was issued already, so
152 ! simply exit the routine. Do not return as a return at
153 ! this point would not allow for proper updating of the
154 ! iteration tracking parameters
155 exit
156 end if
157
158 ! Store previous iteration values
159 xold = x
160 fold = f
161
162 ! Define the right-hand-side for the linear system
163 dir = -fvec
164
165 ! Solve the linear system of equations
166 call solve_lu(jac, ipvt, dir)
167
168 ! Apply the line search if needed
169 if (this%get_use_line_search()) then
170 ! Define the step length for the line search
171 temp = dot_product(dir, dir)
172 if (temp > stpmax) dir = dir * (stpmax / temp)
173
174 ! Apply the line search
175 call limit_search_vector(dir, stpmax)
176 call ls%search(fcn, xold, grad, dir, x, fvec, &
177 fold, f, lib, errmgr)
178 neval = neval + lib%fcn_count
179 else
180 ! No line search - just update the solution estimate
181 x = x + dir
182 call fcn%fcn(x, fvec)
183 f = half * dot_product(fvec, fvec)
184 neval = neval + 1
185 end if
186
187 ! Check for convergence
188 call test_convergence(x, xold, fvec, grad, .true., xtol, &
189 ftol, gtol, check, xcnvrg, fcnvrg, gcnvrg, xnorm, fnorm)
190 if (check) then
191 ! The solution has converged
192 exit
193 else if (gcnvrg) then
194 ! The solution appears to have settled at a point where
195 ! the gradient has a zero slope
196 write(errmsg, '(AI0AE8.3AE8.3)') &
197 "It appears the solution has settled to " // &
198 "a point where the slope of the gradient " // &
199 "is effectively zero. " // new_line('c') // &
200 "Function evaluations performed: ", neval, &
201 "." // new_line('c') // &
202 "Change in Variable: ", xnorm, &
203 new_line('c') // "Residual: ", fnorm
204 call errmgr%report_warning("ns_solve", trim(errmsg), &
205 nl_spurious_convergence_error)
206 end if
207
208 ! Print status
209 if (this%get_print_status()) then
210 call print_status(iter, neval, njac, xnorm, fnorm)
211 end if
212
213 ! Ensure we haven't made too many function evaluations
214 if (neval >= maxeval) then
215 flag = 1
216 exit
217 end if
218 end do
219 end if
220
221 ! Report out iteration statistics
222 if (present(ib)) then
223 ib%iter_count = iter
224 ib%fcn_count = neval
225 ib%jacobian_count = njac
226 ib%gradient_count = 0
227 ib%converge_on_fcn = fcnvrg
228 ib%converge_on_chng = xcnvrg
229 ib%converge_on_zero_diff = gcnvrg
230 end if
231
232 ! Check for convergence issues
233 if (flag /= 0) then
234 write(errmsg, '(AI0AE8.3AE8.3)') "The algorithm failed to " // &
235 "converge. Function evaluations performed: ", neval, &
236 "." // new_line('c') // "Change in Variable: ", xnorm, &
237 new_line('c') // "Residual: ", fnorm
238 call errmgr%report_error("ns_solve", trim(errmsg), &
239 nl_convergence_error)
240 end if
241 end subroutine
242end submodule
