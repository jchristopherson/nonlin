5! ------------------------------------------------------------------------------
6 module subroutine qns_solve(this, fcn, x, fvec, ib, err)
7 ! Arguments
8 class(quasi_newton_solver), intent(inout) :: this
9 class(vecfcn_helper), intent(in) :: fcn
10 real(real64), intent(inout), dimension(:) :: x
11 real(real64), intent(out), dimension(:) :: fvec
12 type(iteration_behavior), optional :: ib
13 class(errors), intent(inout), optional, target :: err
14
15 ! Parameters
16 real(real64), parameter :: zero = 0.0d0
17 real(real64), parameter :: half = 0.5d0
18 real(real64), parameter :: one = 1.0d0
19 real(real64), parameter :: factor = 1.0d2
20
21 ! Local Variables
22 logical :: restart, xcnvrg, fcnvrg, gcnvrg, check
23 integer(int32) :: i, neqn, nvar, flag, lw1, lw2, lw3, neval, iter, &
24 maxeval, jcount, njac
25 real(real64), allocatable, dimension(:) :: work, tau, dx, df, fvold, &
26 xold, s
27 real(real64), allocatable, dimension(:,:) :: q, r, b
28 real(real64) :: test, f, fold, temp, ftol, xtol, gtol, &
29 stpmax, x2, xnorm, fnorm
30 type(iteration_behavior) :: lib
31 class(errors), pointer :: errmgr
32 type(errors), target :: deferr
33 character(len = 256) :: errmsg
34 class(line_search), allocatable :: ls
35
36 ! Initialization
37 restart = .true.
38 xcnvrg = .false.
39 fcnvrg = .false.
40 gcnvrg = .false.
41 neqn = fcn%get_equation_count()
42 nvar = fcn%get_variable_count()
43 neval = 0
44 iter = 0
45 njac = 0
46 ftol = this%get_fcn_tolerance()
47 xtol = this%get_var_tolerance()
48 gtol = this%get_gradient_tolerance()
49 maxeval = this%get_max_fcn_evals()
50 if (present(ib)) then
51 ib%iter_count = iter
52 ib%fcn_count = neval
53 ib%jacobian_count = njac
54 ib%gradient_count = 0
55 ib%converge_on_fcn = fcnvrg
56 ib%converge_on_chng = xcnvrg
57 ib%converge_on_zero_diff = gcnvrg
58 end if
59 if (present(err)) then
60 errmgr => err
61 else
62 errmgr => deferr
63 end if
64 if (this%get_use_line_search()) then
65 if (.not.this%is_line_search_defined()) &
66 call this%set_default_line_search()
67 call this%get_line_search(ls)
68 end if
69
70 ! Input Check
71 if (.not.fcn%is_fcn_defined()) then
72 ! ERROR: No function is defined
73 call errmgr%report_error("qns_solve", &
74 "No function has been defined.", &
75 nl_invalid_operation_error)
76 return
77 end if
78 if (nvar /= neqn) then
79 ! ERROR: # of equations doesn't match # of variables
80 write(errmsg, '(AI0AI0A)') "The number of equations (", neqn, &
81 ") does not match the number of unknowns (", nvar, ")."
82 call errmgr%report_error("qns_solve", trim(errmsg), &
83 nl_invalid_input_error)
84 return
85 end if
86 flag = 0
87 if (size(x) /= nvar) then
88 flag = 3
89 else if (size(fvec) /= neqn) then
90 flag = 4
91 end if
92 if (flag /= 0) then
93 ! One of the input arrays is not sized correctly
94 write(errmsg, '(AI0A)') "Input number ", flag, &
95 " is not sized correctly."
96 call errmgr%report_error("qns_solve", trim(errmsg), &
97 nl_array_size_error)
98 return
99 end if
100
101 ! Local Memory Allocation
102 allocate(q(neqn, neqn), stat = flag)
103 if (flag == 0) allocate(r(neqn, nvar), stat = flag)
104 if (flag == 0) allocate(tau(min(neqn, nvar)), stat = flag)
105 if (flag == 0) allocate(b(neqn, nvar), stat = flag)
106 if (flag == 0) allocate(df(neqn), stat = flag)
107 if (flag == 0) allocate(fvold(neqn), stat = flag)
108 if (flag == 0) allocate(xold(nvar), stat = flag)
109 if (flag == 0) allocate(dx(nvar), stat = flag)
110 if (flag == 0) allocate(s(neqn), stat = flag)
111 if (flag /= 0) then
112 ! ERROR: Out of memory
113 call errmgr%report_error("qns_solve", &
114 "Insufficient memory available.", nl_out_of_memory_error)
115 return
116 end if
117 call qr_factor(r, tau, work, lw1)
118 call form_qr(r, tau, q, work, lw2)
119 call fcn%jacobian(x, b, fv = fvec, olwork = lw3)
120 allocate(work(max(lw1, lw2, lw3)), stat = flag)
121 if (flag /= 0) then
122 ! ERROR: Out of memory
123 call errmgr%report_error("qns_solve", &
124 "Insufficient memory available.", nl_out_of_memory_error)
125 return
126 end if
127
128 ! Test to see if the initial guess is a root
129 call fcn%fcn(x, fvec)
130 f = half * dot_product(fvec, fvec)
131 neval = neval + 1
132 test = zero
133 do i = 1, neqn
134 test = max(abs(fvec(i)), test)
135 end do
136 if (test < ftol) then
137 fcnvrg = .true.
138 end if
139
140 ! Process
141 flag = 0 ! Used to check for convergence errors
142 if (.not.fcnvrg) then
143 ! Determine the maximum line search step
144 stpmax = factor * max(norm2(x), real(nvar, real64))
145
146 ! Main Iteration Loop
147 do
148 ! Update the iteration counter
149 iter = iter + 1
150
151 ! Compute or update the Jacobian
152 if (restart) then
153 ! Compute the Jacobian
154 call fcn%jacobian(x, b, fvec, work)
155 njac = njac + 1
156
157 ! Compute the QR factorization, and form Q & R
158 r = b ! Copy the Jacobian - we'll need it later
159 call qr_factor(r, tau, work)
160 call form_qr(r, tau, q, work)
161
162 ! Reset the Jacobian iteration counter
163 jcount = 0
164 else
165 ! Apply the rank 1 update to Q and R
166 df = fvec - fvold
167 dx = x - xold
168 x2 = dot_product(dx, dx)
169
170 ! Compute S = ALPHA * (DF - B * DX)
171 s = (df - matmul(b, dx))
172 call recip_mult_array(x2, s)
173
174 ! Compute the new Q and R matrices for the rank1 update:
175 ! B' = B + ALPHA * S * DX**T
176 call rank1_update(one, s, dx, b)
177 call qr_rank1_update(q, r, s, dx, work) ! S & DX overwritten
178
179 ! Increment the counter tracking how many iterations have
180 ! passed since the last Jacobian recalculation
181 jcount = jcount + 1
182 end if
183
184 ! Compute GRAD = B**T * F, store in DX
185 call mtx_mult(.true., one, b, fvec, zero, dx)
186
187 ! Store FVEC and X
188 xold = x
189 fvold = fvec
190 fold = f
191
192 ! Solve the linear system: B * DX = -F for DX noting that
193 ! B = Q * R. As such, form -Q**T * F, and store in DF
194 call mtx_mult(.true., -one, q, fvec, zero, df)
195
196 ! Now we have R * DX = -Q**T * F, and since R is upper
197 ! triangular, the solution is readily computed. The solution
198 ! will be stored in the first NVAR elements of DF
199 call solve_triangular_system(.true., .false., .true., r, &
200 df(1:nvar))
201
202 ! Ensure the new solution estimate is heading in a sensible
203 ! direction. If not, it is likely time to update the Jacobian
204 temp = dot_product(dx, df(1:nvar))
205 if (temp >= zero) then
206 restart = .true.
207 if (this%get_print_status()) then
208 call print_status(iter, neval, njac, xnorm, fnorm)
209 end if
210 cycle
211 end if
212
213 ! Apply the line search if needed
214 if (this%get_use_line_search()) then
215 ! Define the step length for the line search
216 temp = dot_product(df(1:nvar), df(1:nvar))
217 if (temp > stpmax) df(1:nvar) = df(1:nvar) * (stpmax / temp)
218
219 ! Apply the line search
220 call limit_search_vector(df(1:nvar), stpmax)
221 call ls%search(fcn, xold, dx, df(1:nvar), x, fvec, fold, &
222 f, lib, errmgr)
223 neval = neval + lib%fcn_count
224 else
225 ! No line search - just update the solution estimate
226 x = x + df(1:nvar)
227 call fcn%fcn(x, fvec)
228 f = half * dot_product(fvec, fvec)
229 neval = neval + 1
230 end if
231
232 ! Test for convergence
233 if (lib%converge_on_zero_diff .and. &
234 this%get_use_line_search()) then
235 call test_convergence(x, xold, fvec, dx, .true., xtol, &
236 ftol, gtol, check, xcnvrg, fcnvrg, gcnvrg, xnorm, fnorm)
237 else
238 call test_convergence(x, xold, fvec, dx, .false., xtol, &
239 ftol, gtol, check, xcnvrg, fcnvrg, gcnvrg, xnorm, fnorm)
240 end if
241 if (.not.check) then
242 ! The solution did not converge, figure out why
243 if (gcnvrg) then
244 ! The slope of the gradient is sufficiently close to
245 ! zero to cause issue.
246 if (restart) then
247 ! We've already tried recalculating a new Jacobian,
248 ! issue a warning
249 write(errmsg, '(AI0AE8.3AE8.3)') &
250 "It appears the solution has settled to " // &
251 "a point where the slope of the gradient " // &
252 "is effectively zero. " // new_line('c') // &
253 "Function evaluations performed: ", neval, &
254 "." // new_line('c') // &
255 "Change in Variable: ", xnorm, &
256 new_line('c') // "Residual: ", fnorm
257 call errmgr%report_warning("nqs_solve", &
258 trim(errmsg), nl_spurious_convergence_error)
259 exit
260 else
261 ! Try computing a new Jacobian
262 restart = .true.
263 end if
264 else
265 ! We have not converged, but we're not stuck with a
266 ! zero slope gradient vector either. Go ahead and
267 ! continue the iteration process without recomputing
268 ! the Jacobian - unless the user dictates a
269 ! recaclulation.
270 if (jcount >= this%m_jDelta) then
271 restart = .true.
272 else
273 restart = .false.
274 end if
275 end if
276 else
277 ! The solution has converged. It's OK to exit
278 exit
279 end if
280
281 ! Print status
282 if (this%get_print_status()) then
283 call print_status(iter, neval, njac, xnorm, fnorm)
284 end if
285
286 ! Ensure we haven't made too many function evaluations
287 if (neval >= maxeval) then
288 flag = 1
289 exit
290 end if
291 end do
292 end if
293
294 ! Report out iteration statistics
295 if (present(ib)) then
296 ib%iter_count = iter
297 ib%fcn_count = neval
298 ib%jacobian_count = njac
299 ib%gradient_count = 0
300 ib%converge_on_fcn = fcnvrg
301 ib%converge_on_chng = xcnvrg
302 ib%converge_on_zero_diff = gcnvrg
303 end if
304
305 ! Check for convergence issues
306 if (flag /= 0) then
307 write(errmsg, '(AI0AE8.3AE8.3)') "The algorithm failed to " // &
308 "converge. Function evaluations performed: ", neval, &
309 "." // new_line('c') // "Change in Variable: ", xnorm, &
310 new_line('c') // "Residual: ", fnorm
311 call errmgr%report_error("qns_solve", trim(errmsg), &
312 nl_convergence_error)
313 end if
314 end subroutine
